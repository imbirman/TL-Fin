function checkRequiredParams() {
	let banks = $('#bankForm input[id^="bank_"]:checked');
	for(i = 0; i < banks.length; i++) {
		let bank_id = parseInt($(banks[i]).val());
		if(([18, 19].includes(bank_id) && card_fz != 44) || [20, 26].includes(bank_id)) // ToDo Временно, пока ПСБ не разрешит включить все ФЗ
			continue;
		if($('[name="CardToBank[' + bank_id + '][format_bg]"]:checked').length == 0) {
			showErrorMsg("Необходимо отметить, требуется ли бумажная БГ", 500);
			hideWaiting();
			return false;
		}
		if(bank_id == 25 && $('[name="CardToBank[' + bank_id + '][format_bg]"]:checked').val() == 1 && 
			(
				$('#CardParam_sber_contacts_phone').val() == "" ||
				$('#CardParam_sber_contacts_person').val() == "" ||
				$('#CardParam_sber_contacts_address').val() == ""
			)
		) {
			showErrorMsg("Необходимо заполнить обязательные поля", 500);
			hideWaiting();
			return false;
		}
		if((bank_id == 18 || bank_id == 19) && $('[name="CardToBank[' + bank_id + '][format_bg]"]:checked').val() == 1 &&
			$('[name="CardToBank[' + bank_id + '][need_delivery]"]:checked').length == 0
		) {
			showErrorMsg("Необходимо заполнить обязательные поля", 500);
			hideWaiting();
			return false;
		}
	}
	return true;
}
